#include <iostream>
#include <time.h>

using namespace std;

int main()
{
	struct tm *tim = new tm;
	time_t tt = time(NULL);
	localtime_s(tim,&tt);
	cout << 1900 + tim->tm_year << "."<< 1 + tim->tm_mon <<"."<< tim->tm_mday << endl;

	system("pause");
	return 0;
}
